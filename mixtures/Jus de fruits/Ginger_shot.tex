\recette{Ginger shot}
\ingredient{1 branche de gingembre}
\ingredient{1 branche de curcuma}
\ingredient{1 ou deux pommes selon les goûts}
\ingredient{1 citron}
\ingredient{50cL d'eau}Laver et éplucher les fruits et épices. Mixer avec l'eau.
\ingredient{}Filtrer à l'aide d'un linge propre ou d'une passoire.
\astuce{3j max au frigo, mais on peut congeler.}
\end{recipe}
