\recette{\OE uf cocotte au curé nantais}
\ingredient{}Pour 4 personnes. Préchauffer le four à 200°C.
\ingredient{200 g. de curé nantais}
\ingredient{120 g. de jambon de Vendée}Détailler le fromage et le jambon en petits dés.
\ingredient{20 g. de beurre}Beurrer l'intérieur de 4 ramequins et répartir le mélange jambon-fromage.
\ingredient{4 cs de crème fraîche}Déposer la cuillère de crème, saler et poivrer.
\ingredient{4 \oe ufs}
\ingredient{Ciboulette}Casser un \oe uf dans chaque ramequin, parsemer de ciboulette.
\sidedish{Mettre 10 minutes au four au bain-marie, servir aussitôt.}
\end{recipe}
