\recette{Tenders sans huile}
\ingredient{1kg de filets de poulet en aiguillettes}
\ingredient{1cs de paprika}
\ingredient{1cc de sel/poivre}
\ingredient{1cs d'oignon moulu}
\ingredient{1cs d'ail moulu}
\ingredient{2cs de soja sucré}
\ingredient{piment (facultatif)}Mariner le poulet avec les épices et laisser reposer.
\ingredient{Cornflakes non sucrés}Ecraser les cornflakes et ajouter 1 cc de chaque épice de la marinade.
\ingredient{10cL de lait}
\ingredient{farine}Préparer un mélange épais sans grumeaux, puis plonger les morceaux de poulet dans les cornflakes assaisonnés et placer sur une plaque allant au four.
\sidedish{25 minutes à 180 en retournant à mi-cuisson.}
\end{recipe}
