\recette{Saucisses Thaï}
\ingredient{pour 4 pers.}
\ingredient{} Préchauffer le four en mode grill.
\ingredient{3 tiges de citronnelle}
\ingredient{8 feuilles de lime surgelées} Retirer les parties dures de la citronnelle et des feuilles de lime. Emincer très finement.
\ingredient{1 grosse échalote} Ciseler l'échalote.
\ingredient{500g de chair à saucisse nature}
\ingredient{20g de pâte de curry rouge}
\ingredient{10g de sucre} Mélanger tous les ingrédients à la main pour bien répartir le curry. Ajouter 1cc rase de sel. Former des petites saucisses.
\sidedish{Disposer sur une plaque et cuire sous le gril du four 5 minutes de chaque côté. Accompagner de riz gluant.} 
\end{recipe}
