\recette{Curry de porc au bambou}
\ingredient{pour 4 pers.}
\ingredient{500g de filet mignon de porc} Emincer la viande.
\ingredient{400g de julienne de bambou en conserve} Egoutter, rincer le bambou. Les mettre dans un récipient et verser de l'eau bouillante dessus. Laisser une minute puis égouter.
\ingredient{1cs de pâte de curry rouge}
\ingredient{50cL de lait de coco} Délayer la pâte de curry dans le lait de coco.
\ingredient{12 feuilles de lime}
\ingredient{1cc rase de sel} 
\ingredient{1cs rase de sucre} Ajouter le sucre, les feuilles, le sel et porter à ébullition.
\sidedish{Ajouter la viande et le bambou et cuire 6 à 7 minutes. Servir avec du riz.}
\end{recipe}
