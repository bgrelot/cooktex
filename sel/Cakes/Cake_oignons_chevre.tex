\recette{Cake oignons chèvre}
\ingredient{250g de farine}
\ingredient{4 oeufs}
\ingredient{1/2 sachet de levure}
\ingredient{10cL de vin blanc}Mélanger farine, oeufs et levure dans un saladier, puis ajouter le vin blanc.
\ingredient{200g d’oignons}
\ingredient{200g de lardons fumés}
\ingredient{150g de fromage de chèvre}Ajouter la garniture et mélanger.
\sidedish{Enfourner à 180° pendant 45 minutes.}
\end{recipe}
