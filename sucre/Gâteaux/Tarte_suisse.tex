\recette{Tarte suisse}
\ingredient{5 cs de farine bien bombées}
\ingredient{2 cs de sucre}
\ingredient{5 cs de lait}
\ingredient{2 cs d'huile}
\ingredient{1/2 sachet de levure}
\ingredient{1 oeuf}Mélanger le tout puis mettre la pâte dans un moule.
\ingredient{3 pommes coupées}Couvrir de pommes, et mettre 25 minutes au four à 200°.
\ingredient{50g de beurre fondu} 
\ingredient{3 cs de sucre}
\ingredient{1 oeuf}Mélanger cette deuxième préparation puis, étaler sur le gâteau.
\sidedish{Remettre au four 20 minutes.}
\end{recipe}
