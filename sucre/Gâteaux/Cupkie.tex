\recette{Cupkie}
\ingredient{1 oeuf}
\ingredient{60g de sucre complet}Blanchir l'oeuf avec le sucre.
\ingredient{150g de farine}
\ingredient{1/2 cc de levure}
\ingredient{60g de beurre mou}Faire un puits dans la farine, ajouter la levure, le beurre et l'oeuf blanchi.
\ingredient{50g de pépites de chocolat}Ajouter les pépites de chocolat et réserver.
\ingredient{100g de chocolat}
\ingredient{75g de beurre}Faire fondre le beurre et le chocolat et mélanger.
\ingredient{2 oeufs}
\ingredient{60g de sucre complet}Batter les oeufs et le sucre et mélanger au chocolat/beurre.
\ingredient{30g de farine}Ajouter la farine au tamis.
\sidedish{8 à 10 minutes à 180. Laisser refroidir 15 minutes avant de démouler.}
\end{recipe}
